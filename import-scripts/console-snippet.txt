(async()=>{let p=1,tp=1,out=[];do{const r=await fetch('https://mlb26.theshow.com/apis/inventory.json?type=mlb_card&page='+p,{credentials:'include'});const d=await r.json();if(d.error){alert('Not logged in: '+d.error);return;}tp=d.total_pages||1;(d.inventory||[]).forEach(i=>{const q=parseInt(i.quantity,10)||0;if(q>0)out.push({uuid:i.uuid,name:i.name,quantity:q});});console.log('inventory page '+p+'/'+tp);p++;}while(p<=tp);const b=new Blob([JSON.stringify({game:'mlb26',cards:out})],{type:'application/json'});const a=document.createElement('a');a.href=URL.createObjectURL(b);a.download='mlb-inventory.json';document.body.appendChild(a);a.click();a.remove();console.log('Exported '+out.length+' cards — now import mlb-inventory.json into the app.');})();
